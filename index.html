async function fazerLogin() { console.log('[DEBUG LOGIN] Iniciando login...'); const user = document.getElementById('username')?.value?.trim(); const pass = document.getElementById('password')?.value?.trim(); if (!user || !pass) { showError('Preencha usuário e senha'); console.log('[DEBUG LOGIN] Campos vazios'); return; } console.log('[DEBUG LOGIN] Usuário:', user); showLoading('Autenticando...'); try { console.log('[DEBUG LOGIN] Enviando POST para Apps Script...'); const res = await fetch(APPS_SCRIPT_URL, { method: 'POST', headers: {'Content-Type': 'text/plain;charset=utf-8'}, body: JSON.stringify({action: 'login', usuario: user, senha: pass}) }); console.log('[DEBUG LOGIN] Resposta recebida:', res.status, res.statusText); if (!res.ok) { throw new Error(`Erro HTTP ${res.status}: ${res.statusText}`); } const data = await res.json(); console.log('[DEBUG LOGIN] Dados do servidor:', data); if (!data.success) { throw new Error(data.message || 'Usuário ou senha inválidos'); } tecnico = {nome: data.nome, usuario: data.usuario, tipo: data.tipo || 'operacao'}; localStorage.setItem('tecnico', JSON.stringify(tecnico)); document.getElementById('corporateHeader').style.display = 'flex'; if (tecnico.tipo === 'admin') { console.log('[DEBUG LOGIN] Usuário admin detectado'); showScreen('adminScreen'); carregarFiltrosAdmin(); carregarLeiturasAdmin(); } else { console.log('[DEBUG LOGIN] Usuário operacao detectado'); showScreen('startScreen'); document.getElementById('nomeTecnico').textContent = tecnico.nome; } } catch (err) { console.error('[DEBUG LOGIN] Erro:', err); showError(err.message || 'Erro ao conectar ao servidor'); } finally { hideLoading(); } }
